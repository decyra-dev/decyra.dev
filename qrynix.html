python · quickstart.py copy 
# Install
pip install qrynix

# Import and initialise
from qrynix import Qrynix

# Define your schema
schema = {
  "sales.transactions": [
    "customer_id", "revenue",
    "region", "quarter"
  ]
}

# Initialise Qrynix
q = Qrynix(schema=schema)

# Ask a business question
sql = q.ask(
  "Top 10 customers by revenue,
   broken down by region"
)

print(sql)
# → Optimized Spark SQL ready to run
